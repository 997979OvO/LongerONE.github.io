使用 Excel 宏进行简单统计

LongerONE 24 Aug 2016

总结下如何使用Excel的宏功能进行简单统计。

  • 环境：Windows7 x64
  • Visual Basic 6.0
  • 软件：WPS 2013专业版

下载地址：WPS2013 专业版及其VBA
提取码：wdtc

不废话，直接上代码，简单暴力：


Private Sub kaoyan()
'
' 已考研统计 Macro
'

Dim b2 As Integer
Dim b3 As Integer

Dim man As Integer
Dim women As Integer

For i = 1 To 500
'读取工作表ONE中第i行，第6列单元格的值
    kaoyanString = Worksheets("ONE").Cells(i, 6).Value
    
    school = Worksheets("ONE").Cells(i, 1).Value
    sex = Worksheets("ONE").Cells(i, 7).Value
    
    Dim kaoyanLen As Integer
    kaoyanLen = Len(kaoyanString)
    
    If (kaoyanLen > 0) Then
    
        If InStr(school, "二本") Then
        'chool中包含“二本”
           b2 = b2 + 1
        End If
        
        If InStr(school, "三本") Then
            b3 = b3 + 1
        End If
        
        If InStr(sex, "男") Then
           man = man + 1
        End If
        
         If InStr(sex, "女") Then
           woman = woman + 1
        End If
                
    End If
    
Next

‘输出统计结果
Worksheets("TWO").Cells(10, 2).Value = b2
Worksheets("TWO").Cells(10, 3).Value = b3

Worksheets("TWO").Cells(13, 2).Value = man
Worksheets("TWO").Cells(13, 3).Value = woman

End Sub
